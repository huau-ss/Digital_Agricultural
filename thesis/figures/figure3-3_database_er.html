    AGRICULTURAL_PRODUCT {
        int id PK
        string name
        string category "vegetable/fruit/livestock/aquatic/grain/other"
        string origin
        string unit
        text description
        boolean is_active
        datetime created_at
        datetime updated_at
    }
    
    PRICE_HISTORY {
        int id PK
        int product_id FK
        date date
        string market_name
        decimal avg_price
        decimal max_price
        decimal min_price
        bigint volume
        string source
        text remarks
    }
    
    CLEANED_PRICE_DATA {
        int id PK
        int product_id FK
        date date
        string market_name
        decimal avg_price
        decimal max_price
        decimal min_price
        bigint volume
        string source
        boolean is_outlier
        string outlier_reason
    }
    
    TRADE_INFO {
        int id PK
        int product_id FK
        int publisher_id FK
        string info_type "supply/demand"
        string title
        text content
        decimal price
        decimal quantity
        string status "active/completed/cancelled"
        datetime created_at
        datetime updated_at
    }
    
    SYSTEM_MESSAGE {
        int id PK
        int user_id FK
        string title
        text content
        string message_type "price_warning/price_alert/system/order"
        string priority "low/medium/high/urgent"
        boolean is_read
        int related_product_id
        string related_product_name
        decimal price_change_percent
        datetime created_at
        datetime read_at
    }
